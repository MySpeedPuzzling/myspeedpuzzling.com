Usage
Reference the sprite once in your markup, then use <use> to render any symbol.
<!-- Option A: external sprite -->
<svg class="stat-icon" width="16" height="16">
  <use href="stat-icons-sprite.svg#stat-avg"/>
</svg>

<!-- Option B: inline sprite (for file:// or offline use) -->
<svg xmlns="http://www.w3.org/2000/svg" style="display:none">
  <symbol id="stat-avg" …>…</symbol>
  <symbol id="stat-med" …>…</symbol>
</svg>

<svg class="stat-icon" width="16" height="16"><use href="#stat-med"/></svg>

.stat-icon { color: var(--stat-accent); }  /* cascades to stroke via currentColor */
